# Find maximum infection rate for all countries 

# Forming a list of all the countries 
countries = list(corona_dataset_aggregated.index)

# Creat an empty list - maximum infections and a for loop
max_infection_rates = []
for country in countries :                                                        # For each country, calculate the derivative and then max rate
    max_infection_rates.append(corona_dataset_aggregated.loc[country].diff().max())      # Directly append the nos. of max rate to created list 
corona_dataset_aggregated['max infection rate'] = max_infection_rates                 # Creating a new column - maximum infections
